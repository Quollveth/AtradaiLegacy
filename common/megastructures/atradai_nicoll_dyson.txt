	entity_offset = { x = -7 y = -7 }
	build_time = 1800
	resources = {
		category = megastructures
		cost = {

			influence = 300
		}

		upkeep = {
			energy = 5
		}
	}

	prerequisites = { "tech_atradai_nicoll_dyson" }

	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_black_hole"
			NOT = { is_star_class = sc_black_hole }
		}
		custom_tooltip = {
			fail_text = "STAR_BLOCKED_BY_MEGA"
			has_star_blocking_mega = yes
		}
		custom_tooltip = {
			fail_text = "ALREADY_HAS_NDB"
			NOT = { has_planet_flag = atradai_NDB }
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_star"
				is_star = yes
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
		}
	}

	ai_weight = {
		factor = 10

		modifier = {
			factor = 0.1
			starbase = { NOT = { has_starbase_size >= starbase_starfortress } }
		}

		modifier = {
			factor = 3
			any_system_planet = {
				is_colony = yes
			}
		}

		modifier = {
			factor = 0.1
			any_neighbor_system = {
				exists = owner
				NOT = {
					owner = { is_same_value = from }
				}
			}
		}
	}

	on_build_complete = {
		set_star_flag = atradai_NDB
	}
}

atradai_nicoll_dyson_3 = {
	entity = "atradai_nicoll_dyson_3_entity"
	construction_entity = "atradai_nicoll_dyson_3_entity"
	portrait = "GFX_evt_atradai_ndb"
	upgrade_from = {
		atradai_nicoll_dyson_2
	}
	prerequisites = { "tech_atradai_nicoll_dyson" }
	place_entity_on_planet_plane = no
	build_time = 2400
	resources = {
		category = megastructures
		cost = {
			alloys = 5000
		}
		upkeep = {
			energy = 400
			alloys = 200
		}
	}
	ai_weight = {
		factor = 10
	}
	on_build_complete = {
		fromfrom.planet = { save_event_target_as = ndb_planet }
		create_fleet = { 
			name = "Nicoll Dyson Beam" 
			settings = { spawn_debris = no } 
			effect = { 
				set_owner = prev.space_owner
				create_ship = { 
					name = "Nicoll Dyson Beam" 
					random_existing_design = atradai_nicoll_dyson
					graphical_culture = mammalian_01
				}
				set_location = {
					target = event_target:ndb_planet
					distance = 0.1
					angle = 0
				}
			}
		}
		if = {
			limit = { fromfrom.planet = { has_planet_flag = just_repaired_ndb } }
			fromfrom.planet = { remove_planet_flag = just_repaired_ndb }
		}
		else = {
			country_event = { id = atra_ndb.103 }
		}
		remove_megastructure = fromfrom
	}
}
