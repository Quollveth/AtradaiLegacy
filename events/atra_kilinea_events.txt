#discovered first kliniea
ship_event = {
	id = atradai_kilinea.001
	title = "atradai_kilinea.001.name"
	desc = "atradai_kilinea.001.desc"
	picture = GFX_evt_kilinea
	show_sound = event_radio_chatter
	location = from
	is_triggered_only = yes

	trigger = {
		owner = { 
			NOT = { 
				has_country_flag = found_first_kilinea
				is_ai = yes 
			} 
		}
		from = {
			has_star_flag = klini_system_d
		}
	}

	immediate = { 
		owner = {
			set_country_flag = found_first_kilinea 
		}
		from = {
			set_star_flag = real_kliniea
			save_global_event_target_as = real_kilinea_target
		}
	}
	option = { name = "atradai_kilinea.100.a" }
}

		every_system_planet = {
			limit = {
